Запрос:

Делаем запрос .
Обязательно в конце ставить точку с запятой ;
Операция в протоколе HTTP : GET . 


SELECT * FORM citys;              /* Вывести всё из таблицы */

SELECT * FORM citys LIMIT 3;      /* Вывести первые три поля из таблицы */

SELECT * FORM citys WHERE id=2;        /* Вывести поле с id №2 */

SELECT * FORM citys ORDER BY id ASC;      /* Отсортировать и вывести по id в возрастающем порядке */

SELECT * FORM citys ORDER BY id DESC;      /* Отсортировать и вывести по id в убывающем порядке */
