Обьекты в JavaScript

 Обьект - это список пар "key:value" .
var emptyObject = {} ; // пустой обьект

Добавление в обьект нового "key:value" .
emptyObject.имя_свойства = 'abc' ; // { имя_свойства : 'abc' }

Удаление из обьекта.
delete emptyObject.имя_свойства ; // {}

Взятие из обьекта по индексу .
emptyObject["ke" + "y"] ; // тоже самое что emptyObject.key ; т.е. вернет содержимое из 'key ' => 'abc'

Получить все имена свойств из обьекта.
var resultArray = Object.keys(twitterObject) ; // вернет [ 'createAt' , 'id' , 'text' , 'user' ]

Проверка наличия имени свойства в обьекте.
tweetObject.hasOwnProperty('user') ; // вернет true или false
