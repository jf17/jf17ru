Unix Time Format

время в секундах от 1 Января 1970 года


package main

import "fmt"
import "time"
import "strconv"

func main() {

    now := time.Now()
    secs := now.Unix() // type Int64
    
    sec_to_string := strconv.FormatInt(secs, 10)
    
    fmt.Println(sec_to_string)
    
    }
