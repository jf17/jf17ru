Apache2 :


// прверка статуса
sudo systemctl status apache2

// перзапуск сервера
sudo service apache2 restart

// запуск вручную сервера
sudo systemctl start apache2

// остановка вручную сервера
sudo systemctl stop apache2

// включить Apache при запуске системы
sudo systemctl enable apache2

// выключить Apache при запуске системы
sudo systemctl disable apache2

// включен автозапуск Apache при запуске системы ?
sudo systemctl is-enabled apache2

// Когда изменили файл юнита, надо заставлять systemd перечитать его, чтобы он использовал уже новый файл
sudo systemctl daemon-reload
