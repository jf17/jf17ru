  $('#BT_send').click(function (){
  var input_value = $('#input_Box').val ()
  
  if (input_value) {
  // is not emtpy

");
    $.ajax({
      url: '/ajax',
      type: "post",

      success: function(r) {
        $('#response').html(r.Quote);
      }
    });

$(".alpha-only").on("input", function(){
  var regexp = /[^0-9A-Z]/g;
  if($(this).val().match(regexp)){
    $(this).val( $(this).val().replace(regexp,'') );
  }
});
